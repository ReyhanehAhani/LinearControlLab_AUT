Q5

zero = tf([1 50] , [1]);
sys_3 = feedback(G*Gc*zero,1);
pole = tf([1] , [1 50]);
sys_4 = feedback(G*Gc*pole,1);
figure(5)
subplot(2, 1, 1);
step(sys_3);
title('add zero');
subplot(2, 1, 2);
step(sys_4);
title('add pole');
figure(6)
subplot(2, 1, 1);
rlocus(sys_3);
title('add zero');
subplot(2, 1, 2);
rlocus(sys_4);
title('add pole');
